# 1. Fetch FIRST_NAME from Worker table using alias name as WORKER_NAME

SELECT 
    FIRST_NAME AS WORKER_NAME
FROM 
    Worker;


# 2. Fetch unique values of DEPARTMENT from Worker table

SELECT DISTINCT 
    DEPARTMENT
FROM 
    Worker;


# 3. Print details for Workers with the first name as "Vipul" and "Satish" (without using OR condition)

SELECT *
FROM 
    Worker
WHERE 
    FIRST_NAME IN ('Vipul', 'Satish');


# 4. Print details of the Workers whose FIRST_NAME contains 'a'

SELECT *
FROM 
    Worker
WHERE 
    FIRST_NAME LIKE '%a%';


# 5. Print details of the admin whose SALARY lies between 100000 and 500000

SELECT *
FROM 
    Worker
WHERE 
    DEPARTMENT = 'Admin'
    AND SALARY BETWEEN 100000 AND 500000;


# 6. Fetch the count of employees working in the department 'Admin'

SELECT 
    COUNT(*) AS TOTAL_EMPLOYEES
FROM 
    Worker
WHERE 
    DEPARTMENT = 'Admin';


#7. Fetch the count of employees working in each department

SELECT 
    DEPARTMENT,
    COUNT(*) AS TOTAL_EMPLOYEES
FROM 
    Worker
GROUP BY 
    DEPARTMENT;


#8. Show the second highest salary from the table

SELECT 
    MAX(SALARY) AS SECOND_HIGHEST_SALARY
FROM 
    Worker
WHERE 
    SALARY < (SELECT MAX(SALARY) FROM Worker);


# 9. Fetch the departments that have less than three people in it

SELECT 
    DEPARTMENT,
    COUNT(*) AS TOTAL_EMPLOYEES
FROM 
    Worker
GROUP BY 
    DEPARTMENT
HAVING 
    COUNT(*) < 3;


#10. Show all departments along with the total number of people in them

SELECT 
    DEPARTMENT,
    COUNT(*) AS TOTAL_EMPLOYEES
FROM 
    Worker
GROUP BY 
    DEPARTMENT;


# 11. Print the name of employees having the highest salary in each department

SELECT 
    FIRST_NAME,
    DEPARTMENT,
    SALARY
FROM 
    Worker W1
WHERE 
    SALARY = (
        SELECT 
            MAX(SALARY)
        FROM 
            Worker W2
        WHERE 
            W1.DEPARTMENT = W2.DEPARTMENT
    );
